fn get_hkcu() -> RegKey {
    RegKey::predef(HKEY_CURRENT_USER)
}

fn get_hklm() -> RegKey {
    RegKey::predef(HKEY_LOCAL_MACHINE)
}

fn check_reg_key_value(path: &str, key: &str, expected_value: u32) -> bool {
    let hklm_path = get_hklm().open_subkey(path);
    if let Ok(reg_path) = hklm_path {
        if let Ok(value) = reg_path.get_value::<u32, _>(key) {
            return value == expected_value;
        }
    }

    let hkcu_path = get_hkcu().open_subkey(path);
    if let Ok(reg_path) = hkcu_path {
        if let Ok(value) = reg_path.get_value::<u32, _>(key) {
            return value == expected_value;
        }
    }
    
    false
}


fn set_reg_key_value(path: &str, key: &str, value: u32, use_hklm: bool) -> Result<(), std::io::Error> {
    let root = if use_hklm { get_hklm() } else { get_hkcu() };
    let (key_handle, _) = root.create_subkey(path)?;
    key_handle.set_value(key, &value)?;
    Ok(())
}

#[tauri::command]
pub fn get_tweaks() -> Vec<Tweak> {
    // This is not async because getting the state should be fast and not block.
    // The PowerShell checks here are synchronous and could be slow, but we accept this for now.
    // A more advanced implementation would use `tauri::async_runtime::spawn` to do this in the background.
    let mut tweaks = get_all_tweaks();
    for tweak in &mut tweaks {
        if tweak.tweak_type == Some("toggle".to_string()) {
            tweak.applied = Some(match tweak.id.as_str() {
                // Performance
                "perf_visual_effects" => check_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Explorer\\VisualEffects", "VisualFXSetting", 2),
                "perf_power_plan" => crate::commands::run_powershell(&tauri::AppHandle::dummy(), "powercfg /getactivescheme").map(|s| s.contains("e9a42b02-d5df-448d-aa00-03f14749eb61")).unwrap_or(false),
                "perf_compatibility_assistant" => check_reg_key_value("SOFTWARE\\Policies\\Microsoft\\Windows\\AppCompat", "DisablePCA", 1),
                "adv_cpu_priority" => check_reg_key_value("SYSTEM\\CurrentControlSet\\Control\\PriorityControl", "Win32PrioritySeparation", 0x26),
                // Network
                "net_nagle_algorithm" => check_reg_key_value("SYSTEM\\CurrentControlSet\\Services\\Tcpip\\Parameters\\Interfaces", "TcpAckFrequency", 1), // Simplified check
                // Privacy
                "privacy_telemetry" => check_reg_key_value("SOFTWARE\\Policies\\Microsoft\\Windows\\DataCollection", "AllowTelemetry", 0),
                "privacy_ad_id" => check_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\AdvertisingInfo", "Enabled", 0),
                "privacy_search_suggestions" => check_reg_key_value("Software\\Policies\\Microsoft\\Windows\\Explorer", "DisableSearchBoxSuggestions", 1),
                "privacy_error_reporting" => check_reg_key_value("SOFTWARE\\Microsoft\\Windows\\Windows Error Reporting", "Disabled", 1),
                // UI
                "ui_show_file_ext" => check_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Explorer\\Advanced", "HideFileExt", 0),
                "ui_dark_mode" => check_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Themes\\Personalize", "AppsUseLightTheme", 0),
                "ui_sticky_keys" => check_reg_key_value("Control Panel\\Accessibility\\StickyKeys", "Flags", 506),
                // Advanced
                "adv_disable_win_update" => crate::commands::run_powershell(&tauri::AppHandle::dummy(), "sc.exe qc wuauserv").map(|s| s.contains("START_TYPE   : 4   DISABLED")).unwrap_or(false),
                "adv_remote_registry" => crate::commands::run_powershell(&tauri::AppHandle::dummy(), "sc.exe qc RemoteRegistry").map(|s| s.contains("START_TYPE   : 4   DISABLED")).unwrap_or(false),
                "adv_remote_management" => crate::commands::run_powershell(&tauri::AppHandle::dummy(), "sc.exe qc WinRM").map(|s| s.contains("START_TYPE   : 4   DISABLED")).unwrap_or(false),
                
                _ => false,
            });
        }
    }
    tweaks
}

pub async fn apply_tweak(app: &AppHandle, id: &str, value: bool) -> Result<(), String> {
    match id {
        // Performance
        "perf_visual_effects" => set_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Explorer\\VisualEffects", "VisualFXSetting", if value { 2 } else { 1 }, false).map_err(|e| e.to_string())?,
        "perf_power_plan" => {
            let guid = if value { "e9a42b02-d5df-448d-aa00-03f14749eb61" } else { "381b4222-f694-41f0-9685-ff5bb260df2e" }; // Ultimate vs Balanced
            run_command(app, "powercfg", vec!["/setactive", guid]).await.map(|_| ())?;
        },
        "perf_compatibility_assistant" => set_reg_key_value("SOFTWARE\\Policies\\Microsoft\\Windows\\AppCompat", "DisablePCA", if value { 1 } else { 0 }, true).map_err(|e| e.to_string())?,
        "adv_cpu_priority" => set_reg_key_value("SYSTEM\\CurrentControlSet\\Control\\PriorityControl", "Win32PrioritySeparation", if value { 0x26 } else { 0x2 }, true).map_err(|e| e.to_string())?,
        
        // Network (This is a simplified script. A real implementation would iterate all network interfaces)
        "net_nagle_algorithm" => {
            let script = if value { // Disable Nagle
                "Get-NetAdapter | ForEach-Object { Set-ItemProperty -Path \"HKLM:\\SYSTEM\\CurrentControlSet\\Services\\Tcpip\\Parameters\\Interfaces\\$($_.InterfaceGuid)\" -Name TcpAckFrequency -Value 1 -Force; Set-ItemProperty -Path \"HKLM:\\SYSTEM\\CurrentControlSet\\Services\\Tcpip\\Parameters\\Interfaces\\$($_.InterfaceGuid)\" -Name TCPNoDelay -Value 1 -Force }"
            } else { // Enable Nagle
                "Get-NetAdapter | ForEach-Object { Remove-ItemProperty -Path \"HKLM:\\SYSTEM\\CurrentControlSet\\Services\\Tcpip\\Parameters\\Interfaces\\$($_.InterfaceGuid)\" -Name TcpAckFrequency -ErrorAction SilentlyContinue; Remove-ItemProperty -Path \"HKLM:\\SYSTEM\\CurrentControlSet\\Services\\Tcpip\\Parameters\\Interfaces\\$($_.InterfaceGuid)\" -Name TCPNoDelay -ErrorAction SilentlyContinue }"
            };
            run_powershell(app, script).await.map(|_| ())?;
        }

        // Privacy
        "privacy_telemetry" => set_reg_key_value("SOFTWARE\\Policies\\Microsoft\\Windows\\DataCollection", "AllowTelemetry", if value { 0 } else { 1 }, true).map_err(|e| e.to_string())?,
        "privacy_ad_id" => set_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\AdvertisingInfo", "Enabled", if value { 0 } else { 1 }, false).map_err(|e| e.to_string())?,
        "privacy_search_suggestions" => set_reg_key_value("Software\\Policies\\Microsoft\\Windows\\Explorer", "DisableSearchBoxSuggestions", if value { 1 } else { 0 }, false).map_err(|e| e.to_string())?,
        "privacy_error_reporting" => set_reg_key_value("SOFTWARE\\Microsoft\\Windows\\Windows Error Reporting", "Disabled", if value { 1 } else { 0 }, true).map_err(|e| e.to_string())?,
        
        // UI
        "ui_show_file_ext" => set_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Explorer\\Advanced", "HideFileExt", if value { 0 } else { 1 }, false).map_err(|e| e.to_string())?,
        "ui_dark_mode" => {
            set_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Themes\\Personalize", "AppsUseLightTheme", if value { 0 } else { 1 }, false)?;
            set_reg_key_value("Software\\Microsoft\\Windows\\CurrentVersion\\Themes\\Personalize", "SystemUsesLightTheme", if value { 0 } else { 1 }, false).map_err(|e| e.to_string())?;
        },
        "ui_sticky_keys" => set_reg_key_value("Control Panel\\Accessibility\\StickyKeys", "Flags", if value { 506 } else { 511 }, false).map_err(|e| e.to_string())?,
        
        // Advanced
        "adv_disable_win_update" => run_command(app, "sc.exe", vec!["config", "wuauserv", "start=", if value { "disabled" } else { "auto" }]).await.map(|_| ())?,
        "adv_remote_registry" => run_command(app, "sc.exe", vec!["config", "RemoteRegistry", "start=", if value { "disabled" } else { "auto" }]).await.map(|_| ())?,
        "adv_remote_management" => run_command(app, "sc.exe", vec!["config", "WinRM", "start=", if value { "disabled" } else { "auto" }]).await.map(|_| ())?,
        
        _ => return Err(format!("Unknown tweak ID: {}", id)),
    };
    Ok(())
}
